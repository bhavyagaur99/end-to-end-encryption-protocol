# documentation: https://www.digitalocean.com/community/tutorials/how-to-create-a-self-signed-ssl-certificate-for-nginx-in-ubuntu-20-04-1

# create a empty directory for the certificates and cd into it
mkdir certs && cd certs

# Create a self-signed certificate for the server-s
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout server-s.key -out server-s.crt

# Create a self-signed certificate for the client-a
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout client-a.key -out client-a.crt

# Create a self-signed certificate for the client-b
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout client-b.key -out client-b.crt

# Create a self-signed certificate for the client-c
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout client-c.key -out client-c.crt
